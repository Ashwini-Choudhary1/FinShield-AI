How we compute balance discrepancies (`error_balance_orig`) and velocity using window functions

int_balance_anomalies.sql 
SELECT
    *,
    -- In legitimate transfers, old_balance - amount should equal new_balance.
    -- Any non-zero difference is an exact financial anomaly indicator:
    ROUND(old_balance_orig - amount - new_balance_orig, 2) AS error_balance_orig,
    ROUND(old_balance_dest + amount - new_balance_dest, 2) AS error_balance_dest,
    
    -- Window function: Transaction velocity by sender within the same hour step:
    COUNT(*) OVER (PARTITION BY name_orig, step) AS tx_velocity_1h
FROM {{ ref('stg_transactions') }}
