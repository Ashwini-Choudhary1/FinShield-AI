SQL Window Functions (`COUNT(*) OVER`) 
-- dbt_finshield/models/intermediate/int_balance_anomalies.sql
with staging as (
    select * from {{ ref('stg_transactions') }}
),
anomalies as (
    select
        *,
        -- Exact balance discrepancies (Victim vs Recipient)
        (old_balance_orig - amount) - new_balance_orig as error_balance_orig,
        (old_balance_dest + amount) - new_balance_dest as error_balance_dest,
        
        -- High-Velocity Window Function: Count transactions by sender in a rolling 1-hour window
        COUNT(*) OVER (
            PARTITION BY name_orig, step
        ) as tx_velocity_1h,
        
        -- Flag when account balance is drained to exact zero
        case when old_balance_orig = amount and new_balance_orig = 0 then 1 else 0 end as flag_orig_drained
    from staging
)
select * from anomalies;
